   cd Floptest

┌───────────────────────────────────────────────────────────────────────────┐
│ EXAMPLE COMMANDS                                                          │
└───────────────────────────────────────────────────────────────────────────┘

From the Floptest folder:

# Quick test (console output)
./test_flopcal.sh

# Full calculation with report
./run_flopcal.sh

# Base YOLOv12 (no DINO)
python flopcal.py --yolo-size s

# Compare model sizes
python flopcal.py --yolo-size n  # nano
python flopcal.py --yolo-size s  # small (yours)
python flopcal.py --yolo-size l  # large

# Your config manually
python flopcal.py \
    --yolo-size s \
    --dino-variant vitb16 \
    --dinoversion 3 \
    --integration single

# Generate custom report
python flopcal.py --yolo-size s --save-report my_report.md

# Get help
python flopcal.py --help
